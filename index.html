Why AMD?

  • No globals!
  • Modules can be downloaded asynchronously and concurrently
  • Dependency management
  • Simplify the build process

AMD Implementations

  • Curl
  • Dojo Loader
  • RequireJS

require()


require(
  configuration, // (optional; object) 
  dependencies,  // (optional; array of strings) 
  callback       // (optional; function) 
)

require() example


require([ "my/app", "other/lib" ], function(app, otherLib){
  // do something with app and otherLib...
});

require() with a config


require({   
  baseUrl: "js/",   
  packages: [     
    { name: "dojo", location: "lib/dojo" },     
    { name: "dijit", location: "lib/dijit" },     
    { name: "dojox", location: "lib/dojox" },     
    { name: "my", location: "my" },     
    { name: "util", location: "util" },     
    { name: "external", location: "http://foo.com/external" }   
  ] 
},
[ "my/app", "dojo/_lang" ], 
function(app, lang){
  // do something with app and lang...
});

define()


define(
  moduleId,      // (optional; string) optional, but DON'T use !!!
  dependencies,  // (optional; array of strings)
  factory        // (function or value)
)

define() example: app/addOne


define(
  [],
  function(){
    var a = 1;
    var addOne = function(num){
      return a + num;
    };

    return addOne;
  }
)

require() using app/addOne


require(
  ['app/addOne'],
  function(addOne){
    console.log(addOne(2)); // 3
  }
)

define() example: app/utils


define(
  [],
  function(){
    var a = 1;
    var utils = {
      
      addOne: function(num){
        return num + a;
      },
      add: function(num1, num2){
        return num1 + num2;
      }

    };
  
    return utils;
  }
)

require() using app/utils


require(
  ['app/utils'],
  function(utils){
    console.log(utils.addOne(2)); // 3
    console.log(utils.add(1,2)); // 3
  }
)

define() just a static/cached object


define(
  {
    a: 1
  }
)
